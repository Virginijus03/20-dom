function tortogGabalas(tortuKiekis, zmoniuKiekis, tortoSvoris) {
  const gabaliukas = Math.floor(
    (tortuKiekis * tortoSvoris * 1000) / zmoniuKiekis
  );
  console.log(gabaliukas);
}
tortogGabalas(3, 17, 1);
